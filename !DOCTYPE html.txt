    <script type="module">
        import { initializeApp } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-app.js";
        import { getAuth, signInAnonymously, signInWithCustomToken } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-auth.js";
        import { getFirestore, doc, setDoc, serverTimestamp } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-firestore.js";

        const firebaseConfig = JSON.parse(__firebase_config);
        const app = initializeApp(firebaseConfig);
        const auth = getAuth(app);
        const db = getFirestore(app);
        const appId = typeof __app_id !== 'undefined' ? __app_id : 'mlbb-shop-storm';

        const packages = [
            { id: 1, dia: 88, price: 3000, hot: false, type: 'dia' },
            { id: 2, dia: 150, price: 5000, hot: false, type: 'dia' },
            { id: 3, dia: 300, price: 10000, hot: false, type: 'dia' },
            { id: 4, dia: 900, price: 30000, hot: false, type: 'dia' },
            { id: 5, dia: 1500, price: 50000, hot: false, type: 'dia' },
            { id: 6, dia: 3000, price: 70000, oldPrice: 100000, hot: true, type: 'dia' },
            { id: 7, dia: "Weekly Pass", price: 3500, hot: true, type: 'pass' }
        ];

        window.selectedMethod = 'Wave';

        const initAuth = async () => {
            if (typeof __initial_auth_token !== 'undefined' && __initial_auth_token) {
                await signInWithCustomToken(auth, __initial_auth_token);
            } else {
                await signInAnonymously(auth);
            }
        };
        initAuth();

        window.selectPackage = (id) => {
            const pkg = packages.find(p => p.id === id);
            window.selectedPkg = pkg;
            document.getElementById('summaryText').innerText = `${pkg.dia} 💎 - ${pkg.price.toLocaleString()} Ks`;
            document.getElementById('page1').classList.add('hidden');
            document.getElementById('page2').classList.remove('hidden');
            window.scrollTo({ top: 0, behavior: 'smooth' });
        };

        window.goBack = () => {
            document.getElementById('page1').classList.remove('hidden');
            document.getElementById('page2').classList.add('hidden');
        };

        window.submitOrder = async () => {
            const gid = document.getElementById('gameId').value;
            const sid = document.getElementById('serverId').value;
            const name = document.getElementById('senderName').value;
            const trans = document.getElementById('lastFive').value;

            if (!gid || !sid || !name || !trans) {
                showToast("⚠️ အချက်အလက်များ ပြည့်စုံစွာဖြည့်ပါ");
                return;
            }

            try {
                const orderId = `STORM-${Date.now()}`;
                const orderRef = doc(db, 'artifacts', appId, 'public', 'data', 'orders', orderId);
                await setDoc(orderRef, {
                    orderId, gameId: gid, serverId: sid, item: window.selectedPkg.dia,
                    price: window.selectedPkg.price, method: window.selectedMethod, 
                    sender: name, transId: trans, status: 'pending', createdAt: serverTimestamp()
                });
                
                showToast("⚡ အချက်လက်စစ်ဆေးနေပါသည်။ မှန်ကန်ပါက ၃မိနစ်မှ - ၅မိနစ်အတွင်း လူကြီးမင်းရဲ့ MLBB အကောင့်ထဲကို စိန်များရောက်ရှိမည်ဖြစ်ပါသည်။", 6000);
                setTimeout(goBack, 6500);
            } catch (err) { 
                showToast("❌ တစ်ခုခုမှားယွင်းနေပါသည်။"); 
            }
        };

        const grid = document.getElementById('packageGrid');
        packages.forEach(pkg => {
            const div = document.createElement('div');
            div.onclick = () => selectPackage(pkg.id);
            div.className = `diamond-card p-6 cursor-pointer relative group border border-white/5`;
            
            const weeklyPassImg = "https://cdn.qukasoft.com/f/990724/cG96YWFtNG0vcUp3ZUdGckg4OG5icmdQYmNFPQ/p/mlbb-weekly-olmos-pass-uz-74928134-sw320sh160.png";
            const normalDiaImg = "https://wpmisc.com/wp-content/uploads/2025/07/diamond-karung-fcfb-original.png";

            div.innerHTML = `
                ${pkg.hot ? `<span class="absolute top-0 right-0 bg-red-600 text-[10px] font-black px-4 py-1 rounded-tr-2xl rounded-bl-2xl shadow-lg z-10 animate-pulse">HOT</span>` : ''}
                <div class="flex flex-col items-center">
                    <img src="${pkg.type === 'pass' ? weeklyPassImg : normalDiaImg}" class="w-24 h-24 mb-3 object-contain drop-shadow-[0_0_15px_rgba(59,130,246,0.5)] transition-transform group-hover:scale-110">
                    <p class="text-white font-black text-2xl text-stroke italic">${pkg.dia.toLocaleString()}</p>
                    <p class="text-[10px] text-blue-400 font-black uppercase tracking-widest">${pkg.type === 'pass' ? 'Weekly Pass' : 'Diamonds'}</p>
                </div>
                <div class="mt-6">
                    ${pkg.oldPrice ? `<p class="text-xs strike-red text-slate-500 font-bold text-center mb-1">${pkg.oldPrice.toLocaleString()} Ks</p>` : ''}
                    <div class="w-full py-3 rounded-xl bg-blue-600 group-hover:bg-blue-500 text-center font-black text-sm transition-colors shadow-lg">
                        ${pkg.price.toLocaleString()} Ks
                    </div>
                </div>
            `;
            grid.appendChild(div);
        });

        function showToast(msg, duration = 3000) {
            const t = document.getElementById('toast');
            t.innerText = msg;
            t.classList.remove('opacity-0', 'translate-y-10');
            t.classList.add('opacity-1', 'translate-y-0');
            setTimeout(() => {
                t.classList.add('opacity-0', 'translate-y-10');
                t.classList.remove('opacity-1', 'translate-y-0');
            }, duration);
        }
    </script>
